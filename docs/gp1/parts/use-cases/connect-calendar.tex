\begin{usecase}{Connect Calendar}
  \ucbasicinfo{HIGH}{Regular}
  \ucshortdescription{This UC allows the user to connect their external calendars to our system.}
  \uctrigger{This UC is triggered when the user selects the option to connect external calendars.}
  \ucactors{User}{CalDAV}
  \ucpreconditions{User must be logged in}
  \ucrelationships{N/A}{N/A}{N/A}
  \ucinputsoutputs{
    \begin{itemize}
      \item \textbf{CalDAV login credentials} (Source: User)
    \end{itemize}
  }{
    \begin{itemize}
      \item \textbf{Calendar data sync status} (Destination: System)
    \end{itemize}
  }
  \ucmainflow{
    \begin{enumerate}
      \item The user clicks the "Connect Calendar" option in the app.
        \ucinfo{The app asks the user to enter their CalDAV credentials.}
      \item The system syncs the external calendar with the user's main calendar.
        \ucinfo{User is able to see his calendars in the app all together.}
    \end{enumerate}
  }
  \ucalternateflows{
    \begin{itemize}
      \item \textbf{Sync Failure:} If syncing fails due to network issues, the system prompts the user to try again or manually refresh.
    \end{itemize}
  }
  \ucconclusion{The UC ends when the user successfully connects and syncs their external calendar with the system.}
  \ucpostconditions{The system has access to the user's external calendar, and events are synced and displayed in the user's main calendar.}
  \ucspecialrequirements{The system must handle multiple calendars efficiently.}
\end{usecase}
